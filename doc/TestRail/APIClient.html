def initialize(connection_data)
  base_url = connection_data[:url]
  unless base_url.match(/\/$/)
    base_url += '/'
  end
  @url      = base_url + 'index.php?/api/v2/'
  @user     = connection_data[:username]
  @password = connection_data[:password]
end
